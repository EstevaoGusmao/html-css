[ height ] = altura [ width ] = largura [ border ] = borda [ padding ] = acolchoamento / preenchimento [ margin ] = margem [ outline ] = contorno / traçado [ BOX-LEVEL ] = quebra para a próxima linha, ocupa o tamanho da tela a largura inteira e depois que termina, ele pula para a linha de baixo Exemplo: <div>, <h1> - <h6>, <p>, <main>, <header>, <nav>, <article>, <aside>, <footer>, <form> e <video> [ INLINE-LEVEL ] não quebra a linha, não ocupa o tamanho inteiro e dá continuidade ao conteúdo na mesma linha Exemplo: <span>, <a>, <code>, <small>, <strong>, <em>, <sup> - <sub>, <label>, <button>, <input> e <select>
